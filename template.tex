% Set the margin for the page
\usepackage[margin=14.11mm]{geometry}

% For skipping the indentation for newlines/new paragraphs (which LaTeX does by default)
\usepackage[parfill]{parskip}


\renewcommand{\section}[1]{
    \textsc{#1}
    \vspace*{-10pt} \\ \hspace*{-5pt} 
    \hrulefill \\
    \vspace*{-15pt}
}

\newcommand{\smallbullet}{
    \small$\bullet$
}

% Use this for lists with small padding and many items, if it
% will be placed directly below sections
\newenvironment{bullet-list-major}{
    \begin{list}{
        \smallbullet
    }{
        \setlength\leftmargin{15pt}\topsep 0pt \itemsep -4pt
    }
} {
    \end{list}
}

% Use this for lists with large padding and fewer items, if it
% will be placed directly below sections
\newenvironment{spaced-bullet-list-major}{
    \begin{list}{
        \smallbullet
    }{
        \setlength\leftmargin{15pt}\topsep 0pt
    }
} {
    \end{list}
}

% Use the next two functions for blocks within a section which contains subpoints.
% Renders as:
%   Company name                                                    2019 - 2020
%   Some description (in italic) about the company or the project here
%   * Point 1
%   * Point 2
%   * Point 3
%
% The first two lines come from "bigblock" and the points come from "bullet-list-minor"
\newcommand{\bigblock}[3]{
    {
        \hspace*{5pt} \textbf{#1} \hfill #2 \newline
        \hspace*{5pt} \textit{#3}
    }
}

\newenvironment{bullet-list-minor}{
    \begin{list}{
        \smallbullet
    }{
        \setlength\leftmargin{30pt}\topsep 0pt \itemsep -3pt
    }
} {
    \end{list}
}
